**ClaudeFromHere.cpp changes:**

1. **Add static helper `FindExecutable`** (place it as a private method in `CClaudeFromHere` or as a file-scope static function above `_LaunchClaude`). Per D-09, D-10, and RESEARCH Pattern 2:

```cpp
static BOOL FindExecutable(PCWSTR exeName, PCWSTR appPathsSubkey, PWSTR szOut, DWORD cchOut)
{
    // Stage 1: SearchPathW (PATH, includes %LOCALAPPDATA%\Microsoft\WindowsApps)
    if (SearchPathW(nullptr, exeName, nullptr, cchOut, szOut, nullptr))
        return TRUE;

    // Stage 2: HKCU App Paths (per RESEARCH Pitfall 2 — wt.exe Store install is HKCU)
    DWORD cb = cchOut * sizeof(WCHAR);
    if (RegGetValueW(HKEY_CURRENT_USER, appPathsSubkey, nullptr,
            RRF_RT_REG_SZ | RRF_ZEROONFAILURE, nullptr, szOut, &cb) == ERROR_SUCCESS
        && szOut[0])
        return TRUE;

    // Stage 3: HKLM App Paths (winget / system-wide installs)
    cb = cchOut * sizeof(WCHAR);
    if (RegGetValueW(HKEY_LOCAL_MACHINE, appPathsSubkey, nullptr,
            RRF_RT_REG_SZ | RRF_ZEROONFAILURE, nullptr, szOut, &cb) == ERROR_SUCCESS
        && szOut[0])
        return TRUE;

    return FALSE;
}
```

2. **Rewrite `_LaunchClaude`** to replace the two `SearchPathW` calls. The new method body:

   a. **Find wt.exe** via `FindExecutable` with App Paths subkey `SOFTWARE\\Microsoft\\Windows\\CurrentVersion\\App Paths\\wt.exe`. If `FindExecutable` returns FALSE, apply Stage 4 fallback for wt.exe only (per D-10):
   ```cpp
   if (!wtFound) {
       WCHAR szAlias[MAX_PATH] = {};
       ExpandEnvironmentStringsW(
           L"%LOCALAPPDATA%\\Microsoft\\WindowsApps\\wt.exe",
           szAlias, ARRAYSIZE(szAlias));
       if (PathFileExistsW(szAlias)) {
           StringCbCopyW(szWt, sizeof(szWt), szAlias);
           wtFound = TRUE;
       }
   }
   ```
   If still not found, show error dialog (per D-11, D-12, UI-SPEC Surface 2):

   ```
   Buttons: MB_OK
   Body: "Windows Terminal was not found on this machine.\n\nTo install, open Microsoft Store and search for 'Windows Terminal', or run:\n    winget install Microsoft.WindowsTerminal\n\nAfter installing, restart Windows Explorer (or sign out and back in)."
   ```

   b. **Find claude.exe** via `FindExecutable` with App Paths subkey `SOFTWARE\\Microsoft\\Windows\\CurrentVersion\\App Paths\\claude.exe`. No Stage 4 for claude.exe (no execution alias). If not found, show error dialog (per D-11, D-12, UI-SPEC Surface 2):
   ```
   Title: "Claude From Here"
   Icon: MB_ICONERROR
   Buttons: MB_OK
   Body: "Claude Code was not found on this machine.\n\nTo install, run in any terminal:\n    npm i -g @anthropic-ai/claude-code\n\nAfter installing, restart Windows Explorer (or sign out and back in)."
   ```

   c. **Read registry flags** from `HKCU\Software\ClaudeFromHere` (per D-03, D-04, D-07, RESEARCH Pattern 1):
   ```cpp
   WCHAR szModel[256]         = {};
   WCHAR szAllowedTools[1024] = {};
   WCHAR szExtraFlags[2048]   = {};
   DWORD dwVerbose            = 0;

   DWORD cb = sizeof(szModel);
   RegGetValueW(HKEY_CURRENT_USER, L"Software\\ClaudeFromHere", L"Model",
       RRF_RT_REG_SZ | RRF_ZEROONFAILURE, nullptr, szModel, &cb);

   cb = sizeof(dwVerbose);
   RegGetValueW(HKEY_CURRENT_USER, L"Software\\ClaudeFromHere", L"Verbose",
       RRF_RT_REG_DWORD | RRF_ZEROONFAILURE, nullptr, &dwVerbose, &cb);

   cb = sizeof(szAllowedTools);
   RegGetValueW(HKEY_CURRENT_USER, L"Software\\ClaudeFromHere", L"AllowedTools",
       RRF_RT_REG_SZ | RRF_ZEROONFAILURE, nullptr, szAllowedTools, &cb);

   cb = sizeof(szExtraFlags);
   RegGetValueW(HKEY_CURRENT_USER, L"Software\\ClaudeFromHere", L"ExtraFlags",
       RRF_RT_REG_SZ | RRF_ZEROONFAILURE, nullptr, szExtraFlags, &cb);
   ```
   Use `RRF_ZEROONFAILURE` so missing key/values silently produce empty buffers (per D-07: no extra flags = current behavior).
