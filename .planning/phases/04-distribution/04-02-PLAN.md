**Job steps in exact order (per D-13):**

1. **Checkout:** `actions/checkout@v4`

2. **Build C++ DLL:** Two steps — configure and build:
   ```yaml
   - name: Configure CMake
     run: cmake -S . -B build -G "Visual Studio 17 2022" -A x64

   - name: Build DLL
     run: cmake --build build --config Release

   - name: Copy DLL to build root
     run: |
       Copy-Item build\Release\ClaudeFromHere.dll build\ClaudeFromHere.dll
       Copy-Item build\Release\ClaudeFromHere.exe build\ClaudeFromHere.exe
     shell: pwsh
   ```

3. **Build config app (.NET):**
   ```yaml
   - uses: actions/setup-dotnet@v4
     with:
       dotnet-version: '8.x'

   - name: Build config app
     run: dotnet build src/ClaudeFromHereConfig/ClaudeFromHereConfig.csproj --configuration Release --nologo
     shell: pwsh
   ```

4. **Copy assets and pack MSIX:** Use dynamic MakeAppx discovery (Pitfall 2 from research):
   ```yaml
   - name: Copy assets and pack MSIX
     run: |
       Copy-Item assets\claude.ico build\claude.ico
       Copy-Item package\AppxManifest.xml build\AppxManifest.xml
       if (-not (Test-Path build\Assets)) { New-Item -ItemType Directory -Path build\Assets | Out-Null }
       Copy-Item package\Assets\* build\Assets -Recurse -Force

       $makeappx = Get-ChildItem "C:\Program Files (x86)\Windows Kits\10\bin" -Recurse -Filter "makeappx.exe" |
           Where-Object { $_.FullName -match "x64" } |
           Sort-Object FullName -Descending |
           Select-Object -First 1 -ExpandProperty FullName
       if (-not $makeappx) { throw "makeappx.exe not found in Windows Kits" }
       Write-Host "Using MakeAppx: $makeappx"

       & $makeappx pack /o /d package /nv /p build\ClaudeFromHere.msix
     shell: pwsh
   ```

5. **Sign MSIX with Azure Trusted Signing (per D-13):**
   ```yaml
   - uses: azure/login@v2
     with:
       client-id: ${{ secrets.AZURE_CLIENT_ID }}
       tenant-id: ${{ secrets.AZURE_TENANT_ID }}
       subscription-id: ${{ secrets.AZURE_SUBSCRIPTION_ID }}

   - uses: azure/artifact-signing-action@v1
     with:
       endpoint: https://eus.codesigning.azure.net/
       signing-account-name: ${{ secrets.AZURE_SIGNING_ACCOUNT }}
       certificate-profile-name: ${{ secrets.AZURE_SIGNING_PROFILE }}
       files-folder: build
       files-folder-filter: msix
       file-digest: SHA256
       timestamp-rfc3161: http://timestamp.acs.microsoft.com
       timestamp-digest: SHA256
   ```

6. **Install Inno Setup (Pitfall 1 from research):**
   ```yaml
   - name: Install Inno Setup
     run: choco install innosetup --yes --no-progress
     shell: pwsh
   ```

7. **Build installer:**
   ```yaml
   - name: Build installer
     run: |
       $iscc = "C:\Program Files (x86)\Inno Setup 6\ISCC.exe"
       if (-not (Test-Path $iscc)) { throw "Inno Setup not found at: $iscc" }
       & $iscc installer\ClaudeFromHere.iss
     shell: pwsh
   ```

8. **Smoke test (per D-15, D-16):**
   ```yaml
   - name: Smoke test installer
     run: |
       # Silent install
       Start-Process -Wait -FilePath "dist\ClaudeFromHere-Setup.exe" -ArgumentList "/VERYSILENT /SUPPRESSMSGBOXES"

       # Verify MSIX registered
       $pkg = Get-AppxPackage -Name "ClaudeFromHere" -ErrorAction SilentlyContinue
       if (-not $pkg) { throw "MSIX package not registered after install" }
       Write-Host "PASS: Package registered: $($pkg.PackageFullName)"

       # Verify files deployed
       $installDir = "$env:LOCALAPPDATA\ClaudeFromHere"
       $required = @("ClaudeFromHere.dll", "ClaudeFromHere.exe", "ClaudeFromHere.msix", "claude.ico", "AppxManifest.xml")
       foreach ($f in $required) {
         if (-not (Test-Path "$installDir\$f")) { throw "Missing file: $installDir\$f" }
       }
       Write-Host "PASS: All expected files present in $installDir"

       # Verify registry entries exist
       $regPath = "HKCU:\Software\ClaudeFromHere"
       # Registry may or may not exist depending on whether config app was run — skip this check
       # The key verification is MSIX + files

       # Silent uninstall
       $uninstaller = Get-ChildItem "$installDir" -Filter "unins*.exe" -ErrorAction SilentlyContinue | Select-Object -First 1
       if ($uninstaller) {
         Start-Process -Wait -FilePath $uninstaller.FullName -ArgumentList "/VERYSILENT /SUPPRESSMSGBOXES"
       } else {
         Write-Host "WARN: Uninstaller not found — skipping uninstall verification"
       }

       # Verify MSIX unregistered
       Start-Sleep -Seconds 2
       $pkgAfter = Get-AppxPackage -Name "ClaudeFromHere" -ErrorAction SilentlyContinue
       if ($pkgAfter) { Write-Host "WARN: MSIX still registered after uninstall (may need Explorer restart)" }
       else { Write-Host "PASS: Uninstall clean." }

       Write-Host "Smoke test complete."
     shell: pwsh
   ```

   Note: Per D-16, UI automation for context menu is best-effort. The smoke test verifies MSIX registration, file deployment, and uninstall — not visual context menu appearance. Add a comment in the YAML: `# UI automation for context menu verification is best-effort per D-16 — skipped in CI`.

